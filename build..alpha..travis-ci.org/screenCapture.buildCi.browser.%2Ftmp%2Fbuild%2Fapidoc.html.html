    parseString = function (str, a, b) {
  var cb, options, parser;
  if (b != null) {
    if (typeof b === 'function') {
      cb = b;
    }
    if (typeof a === 'object') {
      options = a;
    }
  } else {
    if (typeof a === 'function') {
      cb = a;
    }
    options = {};
  }
  parser = new exports.Parser(options);
  return parser.parseString(str, cb);
}
